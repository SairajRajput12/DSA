class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int,int> mp; 
        int index = 0; 
        for(auto it : nums){
            mp[it] = index; 
            index ++; 
        }

        for(int i = 0; i < nums.size(); i ++){
            if(mp.find(target-nums[i]) != mp.end() and i != mp[target-nums[i]]){
                return {i,mp[target-nums[i]]}; 
            }
        }
        return {-1,-1}; 
    }
};
