
pair<int, int> predecessorSuccessor(TreeNode *root, int key)
{
    // Write your code here.
    int pred = -1;
    int succ = -1;

    TreeNode* temp = root;
    while(temp && temp->data != key){
        if(temp->data > key){
            succ = temp->data;
            temp = temp->left;
        }
        else{
            pred = temp->data;
            temp = temp->right;
        }
    }

    if(temp){
        // predecessor
        TreeNode* leftTree = temp->left;
        while(leftTree){
            pred = leftTree->data;
            leftTree = leftTree->right;
        }

        // successor
        TreeNode* rightTree = temp->right;
        while(rightTree){
            succ = rightTree->data;
            rightTree = rightTree->left;
        }
    }

    return {pred, succ};
}
