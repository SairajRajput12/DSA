#include <bits/stdc++.h> 
class Queue {
public:
int *arr; 
int rear; 
int qfront; 
int size; 

    Queue() {
        // Implement the Constructor
        size = 100001;  
        rear = 0;
        qfront = 0; 
        arr = new int[size]; 
    }

    /*----------------- Public Functions of Queue -----------------*/

    bool isEmpty() {
        // Implement the isEmpty() function 
        if(qfront == rear){
            return true; 
        }
        return false; 
    }

    void enqueue(int x) {
        // Implement the enqueue() function
        if(rear == size){
            cerr << "Queue is full " << endl; 
        }
        else{
            arr[rear] = x; 
            rear ++; 
        }
    }

    int dequeue() {
        // Implement the dequeue() function
        if(qfront == rear){
            return -1; 
        }
        else{
            int ans = arr[qfront]; 
            arr[qfront] = -1; 
            qfront ++; 
            if(qfront == rear){
                qfront = 0; 
                rear = 0; 
            }
            return ans; 
        }
    }

    int front() {
        // Implement the front() function
        if(qfront == rear){
            return -1; 
        }
        else{
            return arr[qfront]; 
        }
    }
};
